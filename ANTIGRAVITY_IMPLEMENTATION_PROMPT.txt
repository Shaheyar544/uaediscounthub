import { NextResponse } from 'next/server'
import { searchItems } from '@/lib/amazon/creators-api'
import { upsertProductFromAmazon } from '@/lib/amazon/sync'
import type { CreatorsConfig } from '@/lib/amazon/creators-api'

function getConfig(): CreatorsConfig {
    return {
        credentialId:     process.env.AMAZON_CREDENTIAL_ID!,
        credentialSecret: process.env.AMAZON_CREDENTIAL_SECRET!,
        partnerTag:       process.env.AMAZON_PARTNER_TAG!,
        marketplace:      'www.amazon.ae',
    }
}

export async function GET(req: Request) {
    const { searchParams } = new URL(req.url)
    const query    = searchParams.get('q')?.trim()
    const sortBy   = (searchParams.get('sort') ?? 'Relevance') as any
    const page     = parseInt(searchParams.get('page') ?? '1')
    const minPrice = searchParams.get('min') ? parseInt(searchParams.get('min')!) : undefined
    const maxPrice = searchParams.get('max') ? parseInt(searchParams.get('max')!) : undefined

    if (!query || query.length < 2) {
        return NextResponse.json({ error: 'Query too short' }, { status: 400 })
    }

    const config = getConfig()

    try {
        const result = await searchItems(config, query, {
            sortBy,
            itemPage: page,
            minPrice,
            maxPrice,
            minReviewsRating: 3,
        })

        // Background: save search results to database (fire and forget)
        Promise.all(
            result.products.map(p => upsertProductFromAmazon(p))
        ).catch(err => console.error('[Search upsert background]', err))

        return NextResponse.json({
            products: result.products.map(p => ({
                asin:            p.asin,
                title:           p.title,
                imageUrl:        p.imageUrl,
                price:           p.price,
                originalPrice:   p.originalPrice,
                discountPercent: p.discountPercent,
                savingsAmount:   p.savingsAmount,
                currency:        p.currency,
                rating:          p.rating,
                reviewCount:     p.reviewCount,
                isPrime:         p.isPrimeEligible,
                isInStock:       p.isInStock,
                brand:           p.brand,
                badge:           p.badge,
                affiliateUrl:    p.affiliateUrl,
                deliveryInfo:    p.deliveryInfo,
            })),
            total: result.totalResults,
            query,
            page,
        })
    } catch (err: any) {
        console.error('[Amazon Search API]', err)
        return NextResponse.json({ error: 'Search failed', details: err.message }, { status: 500 })
    }
}

import { NextResponse } from 'next/server'
import { getItems, getVariations } from '@/lib/amazon/creators-api'
import { upsertProductFromAmazon } from '@/lib/amazon/sync'
import { createClient } from '@/utils/supabase/server'
import type { CreatorsConfig } from '@/lib/amazon/creators-api'

function getConfig(): CreatorsConfig {
    return {
        credentialId:     process.env.AMAZON_CREDENTIAL_ID!,
        credentialSecret: process.env.AMAZON_CREDENTIAL_SECRET!,
        partnerTag:       process.env.AMAZON_PARTNER_TAG!,
        marketplace:      'www.amazon.ae',
    }
}

export async function POST(req: Request) {
    // Admin auth check
    const supabase = await createClient()
    const { data: { user } } = await supabase.auth.getUser()
    if (!user) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })

    const { asins, includeVariations = false } = await req.json()

    if (!Array.isArray(asins) || !asins.length) {
        return NextResponse.json({ error: 'asins array required' }, { status: 400 })
    }
    if (asins.length > 50) {
        return NextResponse.json({ error: 'Max 50 ASINs per import' }, { status: 400 })
    }

    const config  = getConfig()
    const results = []
    const products = await getItems(config, asins)

    for (const product of products) {
        const syncResult = await upsertProductFromAmazon(product)
        results.push({
            asin:      product.asin,
            title:     product.title,
            price:     product.price,
            isNew:     syncResult.isNew,
            productId: syncResult.productId,
            error:     syncResult.error,
        })

        if (includeVariations && product.parentAsin && !syncResult.error) {
            const variations = await getVariations(config, product.parentAsin)
            for (const variant of variations) {
                if (variant.asin === product.asin) continue
                const varResult = await upsertProductFromAmazon(variant)
                results.push({
                    asin:      variant.asin,
                    title:     variant.title,
                    price:     variant.price,
                    isNew:     varResult.isNew,
                    productId: varResult.productId,
                    error:     varResult.error,
                    variant:   true,
                })
                await new Promise(r => setTimeout(r, 200))
            }
        }
        await new Promise(r => setTimeout(r, 150))
    }

    return NextResponse.json({
        results,
        succeeded: results.filter(r => !r.error).length,
        created:   results.filter(r => r.isNew).length,
        failed:    results.filter(r => r.error).length,
    })
}

SECTION 12 — ARCHITECTURE REFERENCE
================================================================================

How the Amazon integration flows:

  User types in search bar
       ↓
  AmazonSearchBar (debounced 400ms)
       ↓
  GET /api/amazon/search?q=...
       ↓
  lib/amazon/creators-api.ts → OAuth token (cached 1hr) → Amazon API call
       ↓
  Results returned to UI as dropdown instantly
       ↓  (background, non-blocking)
  lib/amazon/sync.ts → upsertProductFromAmazon() → Supabase tables updated
  (products, product_prices, price_history, deals auto-created if >10% off)

  Vercel Cron (every 2hrs) → GET /api/amazon/sync
       ↓
  refreshPrices() → reads all products with ASIN from DB → getItems() batch
       ↓
  Price drops detected → checkAndFirePriceAlerts() → Twilio WhatsApp sent

  Admin imports ASIN manually → POST /api/amazon/import
       ↓
  AmazonImportPanel shows result table with new/updated/error status

SECTION 13 — KNOWN LIMITATIONS / NOTES FOR DEVELOPER
================================================================================

1. Amazon Creators API v3.2 is region-based. The EU region credentials (which
   you have) work for amazon.ae, amazon.sa, amazon.co.uk etc. — all from one
   credential set.

2. Rate limit is 1 TPS (1 request per second). The sync.ts file already handles
   this with 1100ms delays between batch calls. Do not remove these delays.

3. OAuth tokens expire after 1 hour. The creators-api.ts module caches the token
   in module memory and auto-refreshes. On Vercel serverless functions, the cache
   resets between cold starts — this is fine, just adds one extra auth call.

4. The SUPABASE_SERVICE_ROLE_KEY bypasses Row Level Security. It is only used in
   lib/amazon/sync.ts which runs server-side only. Never expose it to the client.

5. Vercel free tier supports cron jobs. The vercel.json cron will run every 2
   hours when deployed. For local dev, trigger manually via admin panel.

6. If Amazon returns no results for a search query, the search bar will show
   "No products found" — this is correct behaviour, not a bug.

7. The `asin` column added in Section 3 uses a partial unique index
   (WHERE asin IS NOT NULL) so rows without ASINs (manually added products)
   don't conflict with each other.
